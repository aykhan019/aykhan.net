SELECT YEAR(E.EndDate), COUNT(E.EndDate) AS "Count of Workers Left"
FROM Employees AS E
GROUP BY (E.EndDate)

--/ * 13. Issue the number of employees who have worked less than a year. * /
SELECT COUNT(*)
FROM Employees AS E
WHERE YEAR(E.EndDate)- Year(E.StartDate) < 1
