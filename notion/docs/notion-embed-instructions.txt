```html
            border: 0;
            overflow: auto;
        }
    </style>

    <iframe src="/path/to/source"></iframe>

</html>
```
   - Update the `src` attribute in the `<iframe>` tag with the path to your saved HTML file (e.g., `/notion-pages/module-1.html`).
